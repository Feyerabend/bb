cmake_minimum_required(VERSION 3.13)

# Include the Pico SDK
include($ENV{PICO_SDK_PATH}/external/pico_sdk_import.cmake)

project(pico_particle_system C CXX ASM)

set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)

# Initialize the Pico SDK
pico_sdk_init()

# Add display library
add_library(display STATIC
    display.c
)

target_link_libraries(display
    pico_stdlib
    hardware_spi
    hardware_dma
    hardware_gpio
    hardware_pwm
    hardware_irq
    hardware_sync
)

target_include_directories(display PUBLIC
    ${CMAKE_CURRENT_LIST_DIR}
)

# Particle system executable
add_executable(particle_system
    particle_system.c
)

target_link_libraries(particle_system
    pico_stdlib
    pico_multicore
    display
    hardware_sync
)

# Enable USB output, disable UART
pico_enable_stdio_usb(particle_system 1)
pico_enable_stdio_uart(particle_system 0)

# Create map/bin/hex/uf2 files
pico_add_extra_outputs(particle_system)

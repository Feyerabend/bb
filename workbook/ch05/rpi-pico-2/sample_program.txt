# Sample program demonstrating features

let message = "Pico Display Test";
print(message);

let x = 10;
let y = 20;
print("x = ");
print(x);
print("y = ");
print(y);

if x < y {
    print("x is less than y");
    let sum = x + y;
    print("Sum: ");
    print(sum);
} else {
    print("x is greater or equal to y");
}

print("Counting to 10:");
let i = 0;
while i < 10 {
    print(i);
    i = i + 1;
}

let result = x * y;
print("x * y = ");
print(result);

print("Program complete!");
